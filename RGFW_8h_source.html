172RGFW_window* RGFW_createWindowPointer(char* name, int x, int y, int w, int h, unsigned long args){
173 int singleBufferAttributes[] = {4, 8, 8, 9, 8, 10, 8, None};
174
175 int doubleBufferAttributes[] = {4, 5, 8, 8, 9, 8, 10, 8, None};
176
177 RGFW_window* nWin = (RGFW_window*)malloc(sizeof(RGFW_window));
178
179 nWin->event.droppedFilesCount = 0;
180
181 nWin->srcX = nWin->x = x;
182 nWin->srcY = nWin->y = y;
183 nWin->srcW = nWin->w = w;
184 nWin->srcH = nWin->h = h;
185
186 nWin->srcName = nWin->name = name;
187
188 XInitThreads(); /* init X11 threading*/
189
190 /* init the display*/
191 nWin->display = XOpenDisplay(0);
192 if ((Display *)nWin->display == NULL)
193 return nWin;
194
195 long event_mask = KeyPressMask | KeyReleaseMask | ButtonPressMask | ButtonReleaseMask | PointerMotionMask | StructureNotifyMask; /* X11 events accepted*/
196
197 XVisualInfo *vi = glXChooseVisual((Display *)nWin->display, DefaultScreen((Display *)nWin->display), doubleBufferAttributes);
198
199 if (vi == NULL) /* switch to single buffer if double buffer fails*/
200 vi = glXChooseVisual((Display *)nWin->display, DefaultScreen((Display *)nWin->display), singleBufferAttributes);
201
202 if (RGFW_TRANSPARENT_WINDOW & args)
203 XMatchVisualInfo((Display *)nWin->display, DefaultScreen((Display *)nWin->display), 32, TrueColor, vi); /* for RGBA backgrounds*/
204
205 Colormap cmap = XCreateColormap((Display *)nWin->display, RootWindow((Display *)nWin->display, vi->screen),
206 vi->visual, AllocNone); /* make the cmap from the visual*/
207
208 nWin->glWin = (void*)glXCreateContext((Display *)nWin->display, vi, 0, 1); /* create the GLX context with the visual*/
209
210 /* make X window attrubutes*/
211 XSetWindowAttributes swa;
212
213 swa.colormap = cmap;
214 swa.border_pixel = 0;
215 swa.event_mask = event_mask;
216
217 /* create the window*/
218 nWin->window = (void*)XCreateWindow((Display *)nWin->display, RootWindow((Display *)nWin->display, vi->screen), x, y, w, h,
219 0, vi->depth, InputOutput, vi->visual,
220 (1L << 1) | (1L << 13) | CWBorderPixel | CWEventMask, &swa);
221
222 if (RGFW_NO_RESIZE & args){ /* make it so the user can't resize the window*/
223 XSizeHints *sh = XAllocSizeHints();
224 sh->flags = (1L << 4) | (1L << 5);
225 sh->min_width = sh->max_width = w;
226 sh->min_height = sh->max_height = h;
227
228 XSetWMSizeHints((Display *)nWin->display, (GLXDrawable)nWin->window, sh, ((Atom)40));
229 }
230
231
232 if (RGFW_NO_BOARDER & args){
233 /* Atom vars for no-border*/
234 Atom window_type = XInternAtom((Display *)nWin->display, "_NET_WM_WINDOW_TYPE", False);
235 Atom value = XInternAtom((Display *)nWin->display, "_NET_WM_WINDOW_TYPE_DOCK", False);
236
237 XChangeProperty((Display *)nWin->display, (GLXDrawable)nWin->window, window_type, ((Atom)4), 32, PropModeReplace, (unsigned char *)&value, 1); /* toggle border*/
238 }
239
240 XSelectInput((Display *)nWin->display, (GLXDrawable)nWin->window, swa.event_mask); /* tell X11 what events we want*/
241
242 /* make it so the user can't close the window until the program does*/
243 Atom wm_delete = XInternAtom((Display *)nWin->display, "WM_DELETE_WINDOW", 1);
244 XSetWMProtocols((Display *)nWin->display, (GLXDrawable)nWin->window, &wm_delete, 1);
245
246 /* connect the context to the window*/
247 glXMakeCurrent((Display *)nWin->display, (GLXDrawable)nWin->window, (GLXContext)nWin->glWin);
248
249 /* set the background*/
250 XStoreName((Display *)nWin->display, (GLXDrawable) nWin->window, name); /* set the name*/
251
252 glEnable(0x0BE2); /* Enable blending.*/
253 glBlendFunc(0x0302, 0x0303); /* Set blending function*/
254
255 XMapWindow((Display *)nWin->display, (GLXDrawable)nWin->window); /* draw the window*/
256 XMoveWindow((Display *)nWin->display, (GLXDrawable)nWin->window, x, y); /* move the window to it's proper cords*/
257
258
259 if (RGFW_ALLOW_DND & args){
260 nWin->dnd = 1;
261
262 XdndAware = XInternAtom((Display*)nWin->display, "XdndAware", False);
263 XdndTypeList = XInternAtom((Display*)nWin->display, "XdndTypeList", False);
264 XdndSelection = XInternAtom((Display*)nWin->display, "XdndSelection", False);
265
266 /* client messages */
267 XdndEnter = XInternAtom((Display*)nWin->display, "XdndEnter", False);
268 XdndPosition = XInternAtom((Display*)nWin->display, "XdndPosition", False);
269 XdndStatus = XInternAtom((Display*)nWin->display, "XdndStatus", False);
270 XdndLeave = XInternAtom((Display*)nWin->display, "XdndLeave", False);
271 XdndDrop = XInternAtom((Display*)nWin->display, "XdndDrop", False);
272 XdndFinished = XInternAtom((Display*)nWin->display, "XdndFinished", False);
273
274 /* actions */
275 XdndActionCopy = XInternAtom((Display*)nWin->display, "XdndActionCopy", False);
276 XdndActionMove = XInternAtom((Display*)nWin->display, "XdndActionMove", False);
277 XdndActionLink = XInternAtom((Display*)nWin->display, "XdndActionLink", False);
278 XdndActionAsk = XInternAtom((Display*)nWin->display, "XdndActionAsk", False);
279 XdndActionPrivate = XInternAtom((Display*)nWin->display, "XdndActionPrivate", False);
280 const Atom version = 5;
281
282 XChangeProperty((Display*)nWin->display, (Window)nWin->window,
283 XdndAware, 4, 32,
284 PropModeReplace, (unsigned char*) &version, 1);
285 }
286 else
287 nWin->dnd = 0;
288
289 return nWin;
290}

300RGFW_Event RGFW_checkEvents(RGFW_window* win){
301 RGFW_Event event;
302 XEvent E; /* raw X11 event*/
303
304 unsigned char text_uri_list = 1;
305
306 /* if there is no unread qued events, get a new one*/
307 if (XEventsQueued((Display *)win->display, QueuedAlready) + XEventsQueued((Display *)win->display, QueuedAfterReading))
308 XNextEvent((Display *)win->display, &E);
309
310 /* vars for getting the mouse x/y with query*/
311 int x, y, i;
312 unsigned m, m2;
313 Window root, child;
314
315 XWindowAttributes a;
316
317 if (E.type == KeyPress || E.type == KeyRelease) {
318 /* set event key data*/
319 event.keyCode = XkbKeycodeToKeysym((Display *)win->display, E.xkey.keycode, 0, E.xkey.state & ShiftMask ? 1 : 0); /* get keysym*/
320 event.keyName = XKeysymToString(event.keyCode); /* convert to string */
321 /*translateString(event.keyName, 0);*/
322 }
323
324 event.droppedFilesCount = 0;
325 event.dropX = 0;
326 event.dropY = 0;
327
328 switch (E.type) {
329 case KeyPress:
330 event.type = RGFW_keyPressed;
331 break;
332
333 case KeyRelease:
334 /* check if it's a real key release*/
335 if (XEventsQueued((Display *)win->display, QueuedAfterReading)) { /* get next event if there is one*/
336 XEvent NE;
337 XPeekEvent((Display *)win->display, &NE);
338
339 if (E.xkey.time == NE.xkey.time && E.xkey.keycode == NE.xkey.keycode) /* check if the current and next are both the same*/
340 break;
341 }
342
343 event.type = RGFW_keyReleased;
344 break;
345
346 case ButtonPress:
347 event.button = event.button;
348 event.type = RGFW_mouseButtonPressed;
349 break;
350
351 case ButtonRelease:
352 event.type = RGFW_mouseButtonReleased;
353 event.button = event.button;
354 break;
355
356 case MotionNotify:
357 /* if the x/y changes, fetch the x/y with query*/
358 if (XQueryPointer((Display *)win->display, (GLXDrawable)win->window, &root, &child, &x, &y, &x, &y, &m)){
359 event.x = x;
360 event.y = y;
361 }
362 event.type = RGFW_mousePosChanged;
363 break;
364
365 case ClientMessage:
366 /* if the client closed the window*/
367 if (E.xclient.data.l[0] == (long int)XInternAtom((Display *)win->display, "WM_DELETE_WINDOW", 1))
368 event.type = RGFW_quit;
369
370 else if (win->dnd){
371 if (E.xclient.message_type == XdndEnter) {
372 unsigned long count;
373 Atom* formats = NULL;
374 Bool list = E.xclient.data.l[1] & 1;
375
376 xdnd.source = E.xclient.data.l[0];
377 xdnd.version = E.xclient.data.l[1] >> 24;
378 xdnd.format = None;
379
380 if (xdnd.version > 5)
381 break;
382
383 if (list) {
384 Atom actualType;
385 int actualFormat;
386 unsigned long bytesAfter;
387
388 XGetWindowProperty((Display*)win->display,
389 xdnd.source,
390 XdndTypeList,
391 0,
392 LONG_MAX,
393 False,
394 4,
395 &actualType,
396 &actualFormat,
397 &count,
398 &bytesAfter,
399 (unsigned char**) &formats);
400 }
401 else {
402 formats = (Atom*)malloc(3);
403 count = 0;
404 Atom atom[3];
405 if (E.xclient.data.l[2] != None)
406 formats[count++] = E.xclient.data.l[2];
407 if (E.xclient.data.l[3] != None)
408 formats[count++] = E.xclient.data.l[3];
409 if (E.xclient.data.l[4] != None)
410 formats[count++] = E.xclient.data.l[4];
411 }
412
413 unsigned int i;
414 for (i = 0; i < count; i++) {
415 char* name = XGetAtomName((Display*)win->display, formats[i]);
416 if ((strcmp("text/uri-list", name) == 0) || (strcmp("text/plain", name) == 0))
417 xdnd.format = formats[i];
418
419 free(name);
420 }
421
422 if (list && formats)
423 XFree(formats);
424 }
425 else if (E.xclient.message_type == XdndDrop) {
426 event.type = RGFW_dnd;
427
428 Time time = CurrentTime;
429
430 if (xdnd.version > 5)
431 break;
432
433 if (xdnd.format) {
434 if (xdnd.version >= 1)
435 time = E.xclient.data.l[2];
436
437 XConvertSelection((Display*)win->display,
438 XdndSelection,
439 xdnd.format,
440 XdndSelection,
441 (Window)win->window,
442 time);
443
444
445 }
446 else if (xdnd.version >= 2) {
447 XEvent reply = { ClientMessage };
448 reply.xclient.window = xdnd.source;
449 reply.xclient.message_type = XdndFinished;
450 reply.xclient.format = 32;
451 reply.xclient.data.l[0] = (long)win->window;
452 reply.xclient.data.l[1] = 0;
453 reply.xclient.data.l[2] = None;
454
455 XSendEvent((Display*)win->display, xdnd.source,
456 False, NoEventMask, &reply);
457 XFlush((Display*)win->display);
458 }
459 }
460
461 else if (E.xclient.message_type == XdndPosition) {
462 const int xabs = (E.xclient.data.l[2] >> 16) & 0xffff;
463 const int yabs = (E.xclient.data.l[2]) & 0xffff;
464 Window dummy;
465 int xpos, ypos;
466
467 if (xdnd.version > 5)
468 break;
469
470 XTranslateCoordinates((Display*)win->display,
471 XDefaultRootWindow((Display*)win->display),
472 (Window)win->window,
473 xabs, yabs,
474 &xpos, &ypos,
475 &dummy);
476
477 event.dropX = xpos;
478 event.dropY = ypos;
479
480 XEvent reply = { ClientMessage };
481 reply.xclient.window = xdnd.source;
482 reply.xclient.message_type = XdndStatus;
483 reply.xclient.format = 32;
484 reply.xclient.data.l[0] = (long)win->window;
485 reply.xclient.data.l[2] = 0;
486 reply.xclient.data.l[3] = 0;
487
488 if (xdnd.format)
489 {
490 reply.xclient.data.l[1] = 1;
491 if (xdnd.version >= 2)
492 reply.xclient.data.l[4] = XdndActionCopy;
493 }
494
495 XSendEvent((Display*)win->display, xdnd.source,
496 False, NoEventMask, &reply);
497 XFlush((Display*)win->display);
498 }
499 }
500
501 break;
502
503 case SelectionNotify:
504 if (E.xselection.property == XdndSelection && win->dnd) {
505 char* data;
506 unsigned long result;
507
508 Atom actualType;
509 int actualFormat;
510 unsigned long bytesAfter;
511
512 XGetWindowProperty((Display*)win->display, E.xselection.requestor, E.xselection.property, 0, LONG_MAX, False, E.xselection.target, &actualType, &actualFormat, &result, &bytesAfter, (unsigned char**) &data);
513
514 if (result)
515 event.droppedFiles = RGFW_parseUriList(data, &event.droppedFilesCount);
516
517 if (data)
518 XFree(data);
519
520 if (xdnd.version >= 2) {
521 XEvent reply = { ClientMessage };
522 reply.xclient.window = xdnd.source;
523 reply.xclient.message_type = XdndFinished;
524 reply.xclient.format = 32;
525 reply.xclient.data.l[0] = (long)win->display;
526 reply.xclient.data.l[1] = result;
527 reply.xclient.data.l[2] = XdndActionCopy;
528
529 XSendEvent((Display*)win->display, xdnd.source,
530 False, NoEventMask, &reply);
531 XFlush((Display*)win->display);
532 }
533 }
534
535 break;
536 default:
537 XFlush((Display *)win->display);
538 event.type = 0;
539 break;
540 }
541
542 /* if the key has a weird name, change it*/
543 XKeyboardState keystate;
544 XGetKeyboardControl((Display *)win->display, &keystate);
545 event.ledState = keystate.led_mask;
546
547 if ((win->srcX != win->x) || (win->srcY != win->y)){
548 XMoveWindow((Display *)win->display, (GLXDrawable)win->window, win->x, win->y);
549 win->srcX = win->x;
550 win->srcY = win->y;
551 }
552
553 else if ((win->srcW != win->w) || (win->srcH != win->h)){
554 XResizeWindow((Display *)win->display, (GLXDrawable)win->window, (GLXDrawable)win->w, win->h);
555 win->srcW = win->w;
556 win->srcH = win->h;
557 } else {
558 /* make sure the window attrubutes are up-to-date*/
559 XGetWindowAttributes((Display *)win->display, (GLXDrawable)win->window, &a);
560
561 win->srcX = win->x = a.x;
562 win->srcY = win->y = a.y;
563 win->srcW = win->w = a.width;
564 win->srcH =win->h = a.height;
565 }
566
567 if (win->srcName != win->name){
568 XStoreName((Display *)win->display, (Window)win->window, win->name);
569 win->srcName = win->name;
570 }
571
572 win->event = event;
573
574 return event;
575}

582void RGFW_setIcon(RGFW_window* w, unsigned char* src, int width, int height, int channels){
583 int longCount = 2 + width * height;
584
585 unsigned long* icon = (unsigned long*)malloc(longCount * sizeof(unsigned long));
586 unsigned long* target = icon;
587
588 *target++ = width;
589 *target++ = height;
590
591 int i;
592
593 for (i = 0; i < width * height; i++) {
594 if (channels == 3)
595 *target++ = ((src[i * 3 + 0]) << 16) |
596 ((src[i * 3 + 1]) << 8) |
597 ((src[i * 3 + 2]) << 0) |
598 (0xFF << 24);
599
600 else if (channels == 4)
601 *target++ = ((src[i * 4 + 0]) << 16) |
602 ((src[i * 4 + 1]) << 8) |
603 ((src[i * 4 + 2]) << 0) |
604 ((src[i * 4 + 3]) << 24);
605 }
606
607 Atom NET_WM_ICON = XInternAtom((Display*)w->display, "_NET_WM_ICON", False);
608
609 XChangeProperty((Display*)w->display, (Window)w->window,
610 NET_WM_ICON,
611 6, 32,
612 PropModeReplace,
613 (unsigned char*) icon,
614 longCount);
615
616 free(icon);
617
618 XFlush((Display*)w->display);
619}

621char* RGFW_readClipboard(RGFW_window* w){
622 char* result;
623 unsigned long ressize, restail;
624 int resbits;
625 Atom bufid = XInternAtom((Display*)w->display, "CLIPBOARD", False),
626 fmtid = XInternAtom((Display*)w->display, "STRING", False),
627 propid = XInternAtom((Display*)w->display, "XSEL_DATA", False),
628 incrid = XInternAtom((Display*)w->display, "INCR", False);
629 XEvent event;
630
631 XSelectInput ((Display*)w->display, (Window)w->window, PropertyChangeMask);
632 XConvertSelection((Display*)w->display, bufid, fmtid, propid, (Window)w->window, CurrentTime);
633 do {
634 XNextEvent((Display*)w->display, &event);
635 } while (event.type != SelectionNotify || event.xselection.selection != bufid);
636
637 if (event.xselection.property) {
638 XGetWindowProperty((Display*)w->display, (Window)w->window, propid, 0, LONG_MAX/4, True, AnyPropertyType,
639 &fmtid, &resbits, &ressize, &restail, (unsigned char**)&result);
640
641 if (fmtid == incrid)
642 do {
643 while (event.type != PropertyNotify || event.xproperty.atom != propid || event.xproperty.state != PropertyNewValue) XNextEvent((Display*)w->display, &event);
644
645 XGetWindowProperty((Display*)w->display, (Window)w->window, propid, 0, LONG_MAX/4, True, AnyPropertyType,
646 &fmtid, &resbits, &ressize, &restail, (unsigned char**)&result);
647 } while (ressize > 0);
648 }
649
650 return result;
651}

659int RGFW_isPressedI(RGFW_window* w, int key){
660 XQueryKeymap((Display *)w->display, keyboard); /* query the keymap */
661
662 KeyCode kc2 = XKeysymToKeycode((Display *)w->display, key); /* convert the key to a keycode */
663 return !!(keyboard[kc2 >> 3] & (1 << (kc2 & 7))); /* check if the key is pressed */
664}

763RGFW_Event RGFW_checkEvents(RGFW_window* win){

765
766 int setButton = 0;
767
768 win->event.droppedFilesCount = 0;

771 switch (msg.message) {

773 case WM_QUIT:
774 win->event.type = RGFW_quit;
775 break;
776
777 case WM_KEYUP:
778 win->event.keyCode = msg.wParam;

780 win->event.type = RGFW_keyReleased;
781 break;
782
783 case WM_KEYDOWN:
784 win->event.keyCode = msg.wParam;
785 GetKeyNameTextA(msg.lParam, win->event.keyName, 16);

793 break;

797 win->event.type = RGFW_mouseButtonPressed;
798 break;
799 case WM_RBUTTONDOWN:

801 win->event.type = RGFW_mouseButtonPressed;
802 break;
803 case WM_MBUTTONDOWN:
804 win->event.button = RGFW_mouseMiddle;
805 win->event.type = RGFW_mouseButtonPressed;
806 break;

808 case WM_MOUSEWHEEL:

811 else
812 win->event.button = RGFW_mouseScrollDown;

814 win->event.type = RGFW_mouseButtonPressed;
815 break;
816
817 case WM_LBUTTONUP:
818 win->event.button = RGFW_mouseLeft;
819 win->event.type = RGFW_mouseButtonReleased;
820 break;
821 case WM_RBUTTONUP:
822 win->event.button = RGFW_mouseRight;
823 win->event.type = RGFW_mouseButtonReleased;
824 break;
825 case WM_MBUTTONUP:
826 win->event.button = RGFW_mouseMiddle;
827 win->event.type = RGFW_mouseButtonReleased;

829
830 case WM_DROPFILES: {

835 int i;

848 WCHAR* buffer = (WCHAR*)calloc((size_t) length + 1, sizeof(WCHAR));
849

852
853 free(buffer);
854 }
855
856 DragFinish(drop);
857 }
858 break;
859 default: break;
860 }
861
862 TranslateMessage(&msg);
863 DispatchMessage(&msg);
864 }
865
866 if ((win->srcX != win->x) || (win->srcY != win->y) || (win->srcW != win->w) || (win->srcH != win->h)){
867 SetWindowPos((HWND)win->display, (HWND)win->display, win->x, win->y, win->w, win->h, 0);
868 win->srcX = win->x;
869 win->srcY = win->y;
870 win->srcW = win->w;
871 win->srcH = win->h;
872 } else {
873 /* make sure the window attrubutes are up-to-date*/
874 RECT a;
875
876 if (GetWindowRect((HWND)win->display, &a)){
877 win->srcX = win->x = a.left;
878 win->srcY = win->y = a.top;
879 win->srcW = win->w = a.right - a.left;
880 win->srcH = win->h = a.bottom - a.top;
881 }
882 }
883
884 if (win->srcName != win->name){
885 SetWindowTextA((HWND)win->display, win->name);
886 win->srcName = win->name;
887 }
888
889 win->event.keyCode = 0;
890
891 if ((GetKeyState(VK_CAPITAL) & 0x0001)!=0)
892 win->event.keyCode |= 1;
893 if ((GetKeyState(VK_NUMLOCK) & 0x0001)!=0)
894 win->event.keyCode |= 2;
895 if ((GetKeyState(VK_SCROLL) & 0x0001)!=0)
896 win->event.keyCode |= 3;
897
898
899 if (!IsWindow((HWND)win->display))
900 win->event.type = RGFW_quit;
901
902 return win->event;
903}
